CREATE TABLE CATEGORIES 
(
    ID_CATEGORIES NUMBER NOT NULL,
    NAME VARCHAR(55) NOT NULL,
    SEASON VARCHAR(55) NOT NULL,
    
	CONSTRAINT PK_TAB_CATEGORIES PRIMARY KEY (ID_CATEGORIES)
);

CREATE TABLE PRODUCTS
(
    ID_PRODUCTS NUMBER NOT NULL,
    NAME VARCHAR(55) NOT NULL,
    REFERENCE VARCHAR(55) NOT NULL,
    PRICE DECIMAL(10,3) NOT NULL,
    CATEGORY_ID NUMBER NOT NULL,
    
	CONSTRAINT PK_TAB_PRODUCTS PRIMARY KEY (ID_PRODUCTS)
);

CREATE TABLE ANSWERS
(
    ID_ANSWERS NUMBER NOT NULL,
    NUMBER_OF_QUESTION VARCHAR2(255) NOT NULL,
    ANSWER VARCHAR2(255) NOT NULL,
    
    
	CONSTRAINT PK_TAB_ANSWERS PRIMARY KEY (ID_ANSWERS)
);
